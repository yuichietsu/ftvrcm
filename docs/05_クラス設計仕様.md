# クラス設計仕様

## 概要
ftvrcm のクラス設計は、**拡張性**と**軽量性**のバランスを重視しています。アクション機能は `Action` インターフェースで抽象化し、新しいアクション追加時に既存コードの変更を最小限に抑えます。

**実装例**：
```kotlin
// アプリ起動アクション
class LaunchAppAction(private val packageName: String) : Action {
    override fun execute() {
        val intent = context.packageManager.getLaunchIntentForPackage(packageName)
        context.startActivity(intent)
    }
    
    override fun getDescription(): String = "アプリ起動: $packageName"
    override fun getId(): String = "launch_app_$packageName"
}

// URL 開くアクション
class OpenUrlAction(private val url: String) : Action {
    override fun execute() {
        val intent = Intent(Intent.ACTION_VIEW, Uri.parse(url))
        context.startActivity(intent)
    }
    
    override fun getDescription(): String = "URL を開く: $url"
    override fun getId(): String = "open_url_${url.hashCode()}"
}

// 音量調整アクション
class AdjustVolumeAction(private val volumeChange: Int) : Action {
    override fun execute() {
        audioManager.adjustVolume(volumeChange, AudioManager.FLAG_SHOW_UI)
    }
    
    override fun getDescription(): String = when {
        volumeChange > 0 -> "音量上げ"
        volumeChange < 0 -> "音量下げ"
        else -> "音量不変"
    }
    override fun getId(): String = "adjust_volume_$volumeChange"
}
```

**拡張手順**：
1. `Action` インターフェース実装クラス作成
2. `ActionFactory` に登録
3. 完了（他のコードは変更不要）

### 2. Data Layer

#### 2.1 SharedPreferencesDataSource.kt
```kotlin
class SharedPreferencesDataSource(
    private val context: Context
) {
    private val sharedPrefs = context.getSharedPreferences(
        "ftvrcm_settings", Context.MODE_PRIVATE
    )
    
    // 操作モード
    fun getOperationMode(): OperationMode = 
        when (sharedPrefs.getString("operation_mode", "NORMAL")) {
            "MOUSE" -> OperationMode.MOUSE
            else -> OperationMode.NORMAL
        }
    
    fun setOperationMode(mode: OperationMode) {
        sharedPrefs.edit().putString("operation_mode", mode.name).apply()
    }
    
    // キーマッピング
    fun getKeyMapping(): Map<Int, String> =
        sharedPrefs.getStringSet("key_mapping", emptySet())
            ?.associate { Pair(it.substringBefore(":").toInt(), it.substringAfter(":")) }
            ?: emptyMap()
    
    fun setKeyMapping(mapping: Map<Int, String>) {
        val mappingSet = mapping.map { "${it.key}:${it.value}" }.toSet()
        sharedPrefs.edit().putStringSet("key_mapping", mappingSet).apply()
    }
}
```

**責務**：SharedPreferences 操作の抽象化

```kotlin
class RemoteControlMonitorService : Service() {
    private lateinit var keyboardManager: KeyboardManager
    private lateinit var operationModeManager: OperationModeManager

        super.onCreate()
```

### 4. Utility Classes

#### 4.1 ActionFactory.kt
```kotlin
class ActionFactory(private val context: Context) {
    /**
     * アクション ID からアクションインスタンスを生成
     */
    fun create(actionId: String): Action? {
        return when {
            actionId.startsWith("launch_app_") -> {
                val packageName = actionId.substringAfter("launch_app_")
                LaunchAppAction(context, packageName)
            }
            actionId.startsWith("open_url_") -> {
                val url = actionId.substringAfter("open_url_")
                OpenUrlAction(context, url)
            }
            actionId.startsWith("adjust_volume_") -> {
                val volumeChange = actionId.substringAfter("adjust_volume_").toInt()
                AdjustVolumeAction(context, volumeChange)
            }
            else -> null
        }
    }
}
```

**責座**：アクションインスタンス生成の一元管理

```kotlin
class KeyboardManager(private val context: Context) {

    fun addKeyProcessor(processor: (RemoteKeyEvent) -> Unit) {
        keyProcessors.add(processor)
    }
    
    fun start() {
        // KeyEvent 監視開始
    }
    
    fun stop() {
        // KeyEvent 監視停止
        keyProcessors.clear()
    }
    
    private fun dispatchKeyEvent(keyEvent: RemoteKeyEvent) {
        keyProcessors.forEach { it(keyEvent) }
    }
}
```

```kotlin
class MouseEventProcessor(private val context: Context) {
    
    /**
     * キーイベントをマウスイベントに変換
     */
    fun process(keyEvent: RemoteKeyEvent): Boolean {
        return when (keyEvent.keyCode) {
            KeyEvent.KEYCODE_DPAD_UP -> movePointer(0, -10)
            KeyEvent.KEYCODE_DPAD_DOWN -> movePointer(0, 10)
            KeyEvent.KEYCODE_DPAD_LEFT -> movePointer(-10, 0)
            KeyEvent.KEYCODE_DPAD_RIGHT -> movePointer(10, 0)
            KeyEvent.KEYCODE_DPAD_CENTER -> simulateClick()
            else -> false
        }
    }
    
    private fun movePointer(dx: Int, dy: Int): Boolean {
        // MotionEvent 生成 → InputMethodManager 経由で送信
        return true
    }
    
    private fun simulateClick(): Boolean {
        // ACTION_DOWN → ACTION_UP のシーケンス送信
        return true
    }
}
```

```kotlin
// MainActivity.kt
val dataSource = SharedPreferencesDataSource(this)
val settingsRepository = SettingsRepository(dataSource)
val keyMapperUseCase = KeyMapperUseCase(settingsRepository)
val actionFactory = ActionFactory(this)

    settingsRepository, keyMapperUseCase, actionFactory
)
```
