enum class OperationMode {
    NORMAL,      // 通常操作
    MOUSE;       // マウス操作
    
    fun toggle(): OperationMode = when (this) {
        NORMAL -> MOUSE
        MOUSE -> NORMAL
    }
}

### 2. Data Layer

#### 2.1 SharedPreferencesDataSource.kt
```kotlin
class SharedPreferencesDataSource(
    private val context: Context
) {
    private val sharedPrefs = context.getSharedPreferences(
        "ftvrcm_settings", Context.MODE_PRIVATE
    )
    
    // 操作モード
    fun getOperationMode(): OperationMode = 
        when (sharedPrefs.getString("operation_mode", "NORMAL")) {
            "MOUSE" -> OperationMode.MOUSE
            else -> OperationMode.NORMAL
        }
    
    fun setOperationMode(mode: OperationMode) {
        sharedPrefs.edit().putString("operation_mode", mode.name).apply()
    }
    
    // キーマッピング
    fun getKeyMapping(): Map<Int, String> =
        sharedPrefs.getStringSet("key_mapping", emptySet())
            ?.associate { Pair(it.substringBefore(":").toInt(), it.substringAfter(":")) }
            ?: emptyMap()
    
    fun setKeyMapping(mapping: Map<Int, String>) {
        val mappingSet = mapping.map { "${it.key}:${it.value}" }.toSet()
        sharedPrefs.edit().putStringSet("key_mapping", mappingSet).apply()
    }
}
```

**責務**：SharedPreferences 操作の抽象化

---

#### 2.2 SettingsRepository.kt
```kotlin
class SettingsRepository(
    private val dataSource: SharedPreferencesDataSource
) {
    // メモリキャッシュ
    private val operationModeCache = MutableLiveData<OperationMode>()
    
    init {
        operationModeCache.value = dataSource.getOperationMode()
    }
    
    fun getOperationMode(): LiveData<OperationMode> = operationModeCache
    
    fun setOperationMode(mode: OperationMode) {
        dataSource.setOperationMode(mode)
        operationModeCache.value = mode
    }
    
    fun getKeyMapping(): Map<Int, String> = dataSource.getKeyMapping()
    
    fun setKeyMapping(mapping: Map<Int, String>) {
        dataSource.setKeyMapping(mapping)
    }
    
    fun cleanup() {
        // リソース解放
    }
}
```

**責座**：データアクセス層、キャッシング管理

### 3. Service Layer

#### 3.1 RemoteControlMonitorService.kt
```kotlin
class RemoteControlMonitorService : Service() {
    private lateinit var keyboardManager: KeyboardManager
    private lateinit var operationModeManager: OperationModeManager
    
    override fun onCreate() {
        super.onCreate()
        keyboardManager = KeyboardManager(this)
        operationModeManager = OperationModeManager(
            SettingsRepository(SharedPreferencesDataSource(this))
        )
        keyboardManager.start()
    }
    
    override fun onDestroy() {
        super.onDestroy()
        keyboardManager.stop()
    }
    
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        return START_STICKY
    }
    
    override fun onBind(intent: Intent?): IBinder? = null
}
```

**責務**：バックグラウンドリモコン監視

### 4. Utility Classes

#### 4.1 KeyboardManager.kt
```kotlin
class KeyboardManager(private val context: Context) {
    private val keyProcessors = mutableListOf<(RemoteKeyEvent) -> Unit>()
    
    fun addKeyProcessor(processor: (RemoteKeyEvent) -> Unit) {
        keyProcessors.add(processor)
    }
    
    fun start() {
        // KeyEvent 監視開始
    }
    
    fun stop() {
        // KeyEvent 監視停止
        keyProcessors.clear()
    }
    
    private fun dispatchKeyEvent(keyEvent: RemoteKeyEvent) {
        keyProcessors.forEach { it(keyEvent) }
    }
}
```

**責座**：リモコンキーイベント監視と配信

#### 4.2 MouseEventProcessor.kt
```kotlin
class MouseEventProcessor(private val context: Context) {
    
    /**
     * キーイベントをマウスイベントに変換
     */
    fun process(keyEvent: RemoteKeyEvent): Boolean {
        return when (keyEvent.keyCode) {
            KeyEvent.KEYCODE_DPAD_UP -> movePointer(0, -10)
            KeyEvent.KEYCODE_DPAD_DOWN -> movePointer(0, 10)
            KeyEvent.KEYCODE_DPAD_LEFT -> movePointer(-10, 0)
            KeyEvent.KEYCODE_DPAD_RIGHT -> movePointer(10, 0)
            KeyEvent.KEYCODE_DPAD_CENTER -> simulateClick()
            else -> false
        }
    }
    
    private fun movePointer(dx: Int, dy: Int): Boolean {
        // MotionEvent 生成 → InputMethodManager 経由で送信
        return true
    }
    
    private fun simulateClick(): Boolean {
        // ACTION_DOWN → ACTION_UP のシーケンス送信
        return true
    }
}
```

**責座**：マウスイベント処理

### 5. ViewModel

#### 5.1 SettingsViewModel.kt
```kotlin
class SettingsViewModel(
    private val settingsRepository: SettingsRepository,
    private val keyMapperUseCase: KeyMapperUseCase
) : ViewModel() {
    
    val operationMode: LiveData<OperationMode> = 
        settingsRepository.getOperationMode()
    
    val keyMappings = MutableLiveData<Map<Int, String>>()
    
    fun toggleOperationMode() {
        val current = operationMode.value ?: OperationMode.NORMAL
        settingsRepository.setOperationMode(current.toggle())
    }
    
    fun updateKeyMapping(mapping: Map<Int, String>) {
        settingsRepository.setKeyMapping(mapping)
        keyMappings.value = mapping
    }
    
    override fun onCleared() {
        super.onCleared()
        settingsRepository.cleanup()
    }
}
```

**責座**：UI ロジック、状態管理

## 依存性注入の方針

### 手動 DI（推奨）
軽量性重視のため、初版では手動 DI を採用：

```kotlin
// MainActivity.kt
val dataSource = SharedPreferencesDataSource(this)
val settingsRepository = SettingsRepository(dataSource)
val keyMapperUseCase = KeyMapperUseCase(settingsRepository)
val viewModel = SettingsViewModel(
    settingsRepository, keyMapperUseCase
)
```
